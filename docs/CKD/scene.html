Go 端对照实现

import "golang.org/x/crypto/argon2"

salt := make([]byte, 16)
copy(salt, []byte("ckd"))
rootKey := argon2.IDKey([]byte("your-password"), salt, 3, 64*1024, 4, 32)
// rootKey 即为 CKD 的 secret (32 bytes)
